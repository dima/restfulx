      def includes_as_hash(includes = nil)      
        res = case
          when includes.is_a?(Hash)
            includes      
          when includes.nil?
           {}  
          else #Deal with arrays and symbols
            res = [includes].flatten.inject({}) {|include_hash, included| include_hash[included] = {} ; include_hash}
        end
        res
      end
